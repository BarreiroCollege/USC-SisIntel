# Se almacenan en unas variables dos contadores de iteraciones y de iteraciones sin mejora. A continuacin, se inicializa
# la barra de progreso, y se ejecuta el algoritmo en 3 pasos:
# 1. Calcular las soluciones que genera cada hormiga de la colonia.
# 2. Seleccionar la mejor solucin, y calcular su ratio de mejora
# 3. Actualizar la feromona global

# Finalmente, se incrementa el contador de iteraciones sin mejora si el ratio de mejora es inferior o igual al especificado
# y se actualiza la barra de progreso.
# Y ya por ltimo, en funcin de las condiciones de parada, se fuerza a seguir ejecutando el algoritmo si hay unas condiciones
# mnimas especificadas, o se detiene si se llegan a las mximas o a las mximas sin mejora.


    def solucion(self):
        """
        Se devuelve la solucin del camino ptimo de un grafo mediante el algoritmo de colonias de hormigas.
        Dependiendo de la condicin de parada, se llamar a una estructura del algoritmo u otra, siendo la por
        defecto la de un mximo de 1000 iteraciones. Adems, al acabar de encontrar una solucin, se resetean todas
        las matrices temporales, permitiendo ejecutarlo de nuevo cambiando los atributos pblicos.
        :return: solucin ptima encontrada
        """
        iters = 0
        iters_sinmejora = 0

        # Iniciar la abrra de progreso en funcin de la condicin mxima de parada
        iters_max = None
        if self.__parada.iter_max is not None:
            iters_max = self.__parada.iter_max
        elif self.__parada.iter_sinmejoras is not None:
            iters_max = self.__parada.iter_sinmejoras
        pbar = tqdm(total=iters_max, leave=self.__leave)

        while True:
            if self.__parada.iter_max is not None:
                iters += 1
                pbar.update()

            # Se calculan las soluciones para cada hormiga
            sols = self.calcular_soluciones_k()

            # Se selecciona la mejor solucin de las encontradas, incluyendo la actual
            bs, bsc = self.seleccionar_mejor_solucion(sols)
            # Y se calcula el ratio de mejora con respecto a la anterior, actualizando la global tras esto
            if self.__bs is None:
                ratio_mejora = 1
            else:
                ratio_mejora = 1 - (bsc / self.__bsc)
            self.__bs, self.__bsc = bs, bsc

            # Se actualiza la feromona en base a esta solucin encontrada
            self.actualizar_feromona_global()

            # Si se especifica una condicin de parada de iteraciones sin mejora, resetear el contador si el ratio
            # de mejora es superior al especificado (por defecto, es 0)
            if self.__parada.iter_sinmejoras is not None and ratio_mejora > self.__parada.ratio_mejora:
                iters_sinmejora = 0
            # y en caso contrario, incrementar el contador
            elif self.__parada.iter_sinmejoras is not None:
                iters_sinmejora += 1
            # Actualizar la barra de progreso si la condicin de parada mxima es la sin mejoras
            if self.__parada.iter_max is None:
                pbar.update()

            # Si se indica una conidicin de iteraciones mnimas, seguir ejecutando el algoritmo pase lo que pase
            if self.__parada.iter_min is not None and iters < self.__parada.iter_min:
                continue
            # Si se especifica una condicin de parada con ciertas iteraciones sin mejorar y se alcanzan, detener
            if self.__parada.iter_sinmejoras is not None and iters_sinmejora >= self.__parada.iter_sinmejoras:
                break
            # Si se especifica una condicin de parada con un mximo de iteraciones y se alcanzan, detener
            if self.__parada.iter_max is not None and iters >= self.__parada.iter_max:
                break
        # Cerrar la barra de progreso, forzando el completado
        if self.__parada.iter_sinmejoras is not None and pbar.n < pbar.total:
            pbar.update(pbar.total - pbar.n)
        pbar.close()

        # Almacenar la solucin actual en una tupla
        sol = self.__bs, self.__bsc

        # Resetear las matrices de clculo y la solucin obtenida
        self.__init__(self.__g, self.__parada, self.__leave)

        # Devolver la solucin
        return sol

    def calcular_soluciones_k(self):
        """
        Para cada hormiga de la colonia, se calcula una solucin vlida. Para esto, hay dos opciones de calcular los
        nodos siguientes: mediante exploracin dirigida o explotacin. Esto se hace de manera aleatoria, con una
        probabilidad dada por el atributo q0.
        :return: lista de soluciones, una por hormiga
        """
        sols = []
        for k in range(self.m):
            sol = [0]
            # Para cada hormiga, se almacena en una lista las ciudades pendientes de visitar
            disponibles = list(range(1, self.__g.nciudades))
            # Se itera mientras queden ciudades por visitar
            while len(disponibles) > 0:
                # La ciudad i ser la ltima visitada
                i = sol[len(sol) - 1]
                q = random.uniform(0, 1)
                # En funcin de un valor aleatorio, se decide que la ciudad a visitar j ser extraida por
                # explotacin o por exploracin dirigida
                if q < self.q0:
                    j = self.explotacion(disponibles, i)
                else:
                    j = self.exploracion_dirigida(disponibles, i)
                # Se inserta el elemento en la solucin de la hormiga y se elimina de las disponibles
                sol.append(j)
                disponibles.remove(j)
            # Se guarda la solucin en la lista de soluciones, una por hormiga
            sols.append(sol + [0])
        return sols

    def explotacion(self, disponibles, i):
        """
        En la asignacin de caminos, la explotacin consiste en realizar una bsqueda local para encontrar
        el mejor camino. Se tiene en cuenta la distancia y la feromona, pero slo se tendr en cuenta cual es el
        prximo mejor en base a esos dos parmetros.
        :param disponibles: ciudades pendientes de visitar
        :param i: ciudad actual
        :return: ciudad que se ha de visitar
        """
        opciones = []
        for l in disponibles:
            # Para cada ciudad disponible, almacenar un valor indicando feromona y distancia
            opciones.append((l, (self.__tij[i][l] ** self.alpha) * (self.__nij[i][l] ** self.beta)))
        opciones = sorted(opciones, key=lambda o: o[1])
        # Se devuelve la ciudad cuyo valor sea el ms alto
        j = opciones[len(opciones) - 1][0]
        return j


    def exploracion_dirigida(self, disponibles, i):
        """
        En la asignacin de caminos, esta funcin devolver el siguiente nodo para una hormiga dada utilizando
        exploracin dirigda. Esta consiste en asignar unas proabilidades, y en base a una distribucin ordenada,
        decidir de manera aleatoria por probabilidad a cual se avanza.
        :param disponibles: ciudades pendientes por visitar
        :param i: ciudad actual
        :return: ciudad a visitar a continuacin
        """

        # Se calcula la suma de todas las probabilidades
        s = sum((self.__tij[i][l] ** self.alpha) * (self.__nij[i][l] ** self.beta) for l in disponibles)
        opciones = []
        for l in disponibles:
            # Para cada opcin, calcular su probabilidad divida por la total
            opciones.append((l, ((self.__tij[i][l] ** self.alpha) * (self.__nij[i][l] ** self.beta)) / s))
        # Se ordena las soluciones por probabilidad de mayor a menor
        opciones = sorted(opciones, key=lambda o: 1 / o[1])

        # Se genera un nmero aleatorio de "corte"
        m = random.uniform(0, 1)
        j, agregado = None, 0
        for opcion in opciones:
            # Se suma la probabilidad agregada, y se almacena la ciudad a visitar
            agregado += opcion[1]
            j = opcion[0]
            # Si la suma agregada es mayor que el corte, se para de iterar y se devuelve la ciudad encontrada
            if agregado >= m:
                break
        return j
